import java.util.Scanner;

public class Split_MultiSet {
    public static int multiset_split(int S,int k){
        if(S==1)return 0;
      return (int) Math.ceil((double) (S - 1) /(k-1));
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        while(T>0){
            int S = sc.nextInt();
            int k = sc.nextInt();
            System.out.println(multiset_split(S, k));
            T--;
        }
    }
}
